# PRG implies no asserts.
%if defined (PRG)
    %ifndef DEBUG
        %error Cannot define PRG and DEBUG.
    %endif
    %if !defined(NDEBUG) && PRG_NO_NDEBUG == "yes"
        always:
            |Setting NDEBUG=1 because PRG was set.
        
        NDEBUG = 1
    %elif PRG_NO_NDEBUG == "maybe"
        NDEBUG = -nope
    %else
        NDEBUG + 1
    %endif
%endif

    (is_defined

    (body
      (if
        (identifier)
        (body
          (diagnostic
            (restOfLine))))
      (if
        (eq
          (and
            (not
              (is_defined
                (identifier)))
            (identifier))
          (string))
        (body
          (rule
            (always)
            (rule_body
              (build_command
                (command_mod
                  (silent_echo))
                (command
                  (restOfLine)))))
          (assign
            (identifier)
            (restOfLine)))
        (elif_clause
          (restOfLine)
          (body
            (assign
              (identifier)
              (restOfLine))))
        (else_clause
          (body
            (append_assign
              (identifier)
              (restOfLine))))))))
